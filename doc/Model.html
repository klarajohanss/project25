def fetch_order_items(order_id)
    db_connection.execute("SELECT order_items.*, products.namn AS produkt_namn
                           FROM order_items
                           JOIN products ON order_items.produkt_id = products.id
                           WHERE order_items.order_id = ?", [order_id])
end
